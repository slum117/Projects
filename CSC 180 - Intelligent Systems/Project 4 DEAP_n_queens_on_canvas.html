10 pts: Write your code in the cell below to define the "evaFitness" function, which returns the fitness of any given board.¶

  • Noticed that in this case, mutation may generate invalid board, e.g., the board with dupliciate positions. Think about [5, 32, 8, 8, 41, 3, 55, 49]

  • How to exclude those invalid boards from each generation? One way is to add some penalty to the fitness value of invalid boards. In that case, any invalid board will have a very high fitness value (remember that our goal is to find the board with least fitness value). To do that, let's write a function checkDuplicate() to calculate the number of queen pairs in the same position for any given board. Give each duplicate a high penalty (i.e., multiply by 20, 50) and add the penalty to the fitness value.

  • evaFitness() returns the total number of duplicate position pair (with penalty) plus the total number of distinct pairs of queens that attack each other.

def evaFitness(individual):
    
    conflict = 0
    
    for piece in range(len(individual)):
        conflictingValues = list()
        startPosition = individual[piece]
        piecePosition = startPosition
        rowStart = int(individual[piece]/8)*8
        rowEnd = int(individual[piece]/8) * 8 + 8
        #rows
        for i in range(rowStart, rowEnd):
            conflictingValues.append(i)
        #columns
        while(piecePosition > 7):
            piecePosition = piecePosition - 8
            conflictingValues.append(piecePosition)
        piecePosition = startPosition
        while(piecePosition < 56):
            piecePosition = piecePosition + 8
            conflictingValues.append(piecePosition)
        #diagonals
        #upper left
        piecePosition = startPosition
        x,y = startPosition-rowStart, int(startPosition/8)
        while( x > 0 and y > 0):
            piecePosition = piecePosition - 9
            conflictingValues.append(piecePosition)
            x = x - 1
            y = y - 1
        #upper Right
        piecePosition = startPosition
        x,y = 8-(startPosition-rowStart)-1, int(startPosition/8)
        while( x > 0 and y > 0):
            piecePosition = piecePosition - 7
            conflictingValues.append(piecePosition)
            x = x - 1
            y = y - 1
        #bottom Left
        piecePosition = startPosition
        x,y = startPosition-rowStart, 8-int(startPosition/8)-1
        while(x > 0 and y > 0):
            piecePosition = piecePosition + 7
            conflictingValues.append(piecePosition)
            x = x - 1
            y = y - 1
        #bottomRight
        piecePosition = startPosition
        x,y = 8-(startPosition-rowStart)-1, 8-int(startPosition/8)-1
        while(x > 0 and y > 0):
            piecePosition = piecePosition + 9
            conflictingValues.append(piecePosition)
            x = x - 1
            y = y - 1
        
        
        conflictingValues.remove(startPosition)
        conflictedWith = list()
        
        for otherPieces in individual:
            if otherPieces in conflictingValues:
                conflict = conflict + 1
                conflictedWith.append(otherPieces)

        #print(conflictedWith)
        

    conflict = conflict / 2
    return (checkDuplicate(individual) + conflict,)
     
    
# Calculate the number of queen pairs in the same position for any given board
def checkDuplicate(individual):
    dup = 0
    for i in range(len(individual)):
        for j in range(len(individual)):
            if (i != j):
                if (individual[i] == individual[j]):
                    dup = dup + 1

    return dup*100 #50 is the penalty for duplicates

def evaFitness(individual):
    
    conflict = 0
    # check if any individual is above or below the current piece
    #piece is being used as the y value
    for piece in range(len(individual)):
        for otherPiece in range(len(individual)):
            if piece != otherPiece:
                if individual[piece] == individual[otherPiece]:
                    conflict = conflict + 1
    
    #check upper left diagonals
    
    for piece in range(len(individual)):
        x,y = individual[piece],piece
        n,m = 0,0
        temp = 0
        #print(str(x) + " " + str(y))
        if y > 0:
            while n < x and m < y:
                n = n + 1
                m = m + 1
                temp = (individual[piece]+7*n)%8
                check = individual[piece-n]
                if temp == check:
                    conflict = conflict + 1
    #upper right
    for piece in range(len(individual)):
        x,y = 8-individual[piece]-1,piece
        n,m = 0,0
        temp = 0
        #print(str(x) + " " + str(y))
        if y > 0:
            while n < x and m < y:
                n = n + 1
                m = m + 1
                temp = (individual[piece]+9*n)%8
                check = individual[piece-n]
                if temp == check:
                    conflict = conflict + 1
    #lower left
    for piece in range(len(individual)):
        x,y = individual[piece],8-piece-1
        n,m = 0,0
        temp = 0
        #print(str(x) + " " + str(y))
        if y > 0:
            while n < x and m < y:
                n = n + 1
                m = m + 1
                temp = (individual[piece]+7*n)%8
                check = individual[piece+n]
                if temp == check:

                    m = m + 1
                    temp = (individual[piece]+9*n)%8
                    check = individual[piece+n]
                    if temp == check:
                        conflict = conflict + 1
